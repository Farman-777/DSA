#include <iostream>
using namespace std;

int main()
{
int i,j,temp,t;
int a[]={27,23,20,25,16,18,15};
int n=sizeof(a)/sizeof(a[0]);
for(i=0;i<n-1;i++)
{
    temp=a[i];
    t=j;
    for(j=i+1;j<n;j++)
    {
        if(a[j]<temp)
        { temp=a[j]; t=j;}
        
    }
    swap(a[i],a[t]);
    
}
   
for(i=0;i<n;i++)
cout<<a[i]<<" ";
    return 0;
}
