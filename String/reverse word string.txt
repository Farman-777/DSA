#include <bits/stdc++.h>
using namespace std;
void reverseWords(string s)
{
	
	// temporary vector to store all words
	vector<string> tmp;
	string str = "";
	for (int i = 0; i < s.length(); i++)
	{
		if (s[i] == ' ')
		{
			tmp.push_back(str);
			str = "";
		}
		else
			str += s[i];
	}

	// Last word remaining,add it to vector
	tmp.push_back(str);
	
	int i;
	for (i = tmp.size() - 1; i > 0; i--)
		cout << tmp[i] << " ";
	// Last word remaining,print it
	cout << tmp[0] << endl;
}

// Driver Code
int main()
{
	string s = "i like this program very much";
	reverseWords(s);
	return 0;
}

#include <bits/stdc++.h>
using namespace std;
string reverseWords(string s) {
    string result;
    int n=s.length();
    int i=0,j=0;
    string sub;
    while(i<n){
        while(i<n && s[i]==' ') i++;
        if(i>=n)break;
        j=i+1;
        while(j<n && s[j]!=' ')j++;
        sub=s.substr(i,j-i);
        if(result.length()==0) result=sub;
        else{ result=sub+" "+result; }
        i=j+1;
    }
    result[0]=toupper(result[0]);
    result[n-1]=toupper(result[n-1]);
    return result;
    
}
