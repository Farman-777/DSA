#include <iostream>
using namespace std;
#define N 5

int stack1[N];
int stack2[N];

int top1=-1,top2=-1;
int count=0;


void push1(int x)
{
    if(top1==N-1) { cout<<"Stack is full\n"; }
    else{ top1++; stack1[top1]=x; }
}

void push2(int y)
{
    if(top2==N-1) { cout<<"Stack is full\n"; }
    else{ top2++; stack2[top2]=y; }
}

int pop1(){
    if(top1 == -1) { cout<<"stack is empty\n"; }
    else { return stack1[top1--]; }
}

int pop2(){
    if(top2 == -1) { cout<<"stack is empty\n"; }
    else { return stack2[top2--]; }
}

void enqueue( int data)
{
    push1(data);
    count++;
}

void dequeue()
{
    int a,b,i;
    for(i=0;i<count;i++)
    {
        a=pop1();
        push2(a);
    }
    b=pop2();
    cout<<b<<" is the poped Element\n";
    
    count--;
    for(i=0;i<count;i++)
    {
        a=pop2();
        push1(a);
    }
    
}

void display()
{
    int i;
    for(i=0;i<=top1;i++)
    {
        cout<<stack1[i]<<" ";
    }
}


int main()
{
enqueue(5);
enqueue(10);
enqueue(-1);display();cout<<endl;
enqueue(55);
enqueue(6);display();cout<<endl;
dequeue();display();cout<<endl;
dequeue();display();
    return 0;
}
